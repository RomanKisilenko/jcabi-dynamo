59  /**
60   * Mock data in H2 database.

62   * @author Yegor Bugayenko (yegor@tpc2.com)

67  @Immutable
68  @ToString
69  @Loggable(Loggable.DEBUG)
70  @EqualsAndHashCode(of = "jdbc")
71  public final class H2Data implements MkData {

73      /**
74       * Fetcher of rows.
75       */
76      private static final JdbcSession.Handler<Iterable<Attributes>> HANDLER =
77          // @checkstyle AnonInnerLengthCheck (50 lines)
78          new JdbcSession.Handler<Iterable<Attributes>>() {
79              @Override
80              public Iterable<Attributes> handle(final ResultSet rset,
81                  final Statement stmt) throws SQLException {
82                  final Collection<Attributes> items =
83                      new LinkedList<Attributes>();
84                  while (rset.next()) {
85                      items.add(this.fetch(rset));
86                  }
87                  return items;
88              }
89              /**
90               * Convert result set to Attributes.
91               * @param rset Result set
92               * @return Attribs
93               * @throws SQLException If fails
94               */
95              private Attributes fetch(final ResultSet rset) throws SQLException {
96                  final ResultSetMetaData meta = rset.getMetaData();
97                  Attributes attrs = new Attributes();
98                  for (int idx = 0; idx < meta.getColumnCount(); ++idx) {
99                      attrs = attrs.with(
100                         meta.getColumnName(idx + 1),
101                         rset.getString(idx + 1)
102                     );
103                 }
104                 return attrs;
105             }
106         };

165     @Override
166     public Iterable<Attributes> iterate(final String table,
167         final Conditions conds) throws IOException {
168         try {
169             final StringBuilder sql = new StringBuilder("SELECT * FROM ")
170                 .append(table);
171             if (!conds.isEmpty()) {
172                 sql.append(" WHERE ");
173                 Joiner.on(" AND ").appendTo(
174                     sql,
175                     Iterables.transform(conds.keySet(), H2Data.WHERE)
176                 );
177             }
178             JdbcSession session = new JdbcSession(this.connection())
179                 .sql(sql.toString());
180             for (final Condition cond : conds.values()) {
181                 if (cond.getAttributeValueList().size() != 1) {
182                     throw new UnsupportedOperationException(
183                         "at the moment only one value of condition is supported"
184                     );
185                 }
186                 if (!cond.getComparisonOperator()
187                     .equals(ComparisonOperator.EQ.toString())) {
188                     throw new UnsupportedOperationException(
189                         String.format(
190                             "at the moment only EQ operator is supported: %s",
191                             cond.getComparisonOperator()
192                         )
193                     );
194                 }
195                 final AttributeValue val = cond.getAttributeValueList().get(0);
196                 session = session.set(val.getS());
197             }
198             return session.select(H2Data.HANDLER);
199         } catch (SQLException ex) {
200             throw new IOException(ex);
201         }
202     }
203     @Override
204     public void put(final String table, final Attributes attrs)
205         throws IOException {
206         try {
207             JdbcSession session = new JdbcSession(this.connection());
208             for (final AttributeValue value : attrs.values()) {
209                 session = session.set(value.getS());
210             }
211             session.sql(
212                 String.format(
213                     "INSERT INTO %s (%s) VALUES (%s)",
214                     table,
215                     Joiner.on(',').join(attrs.keySet()),
216                     Joiner.on(',').join(Collections.nCopies(attrs.size(), "?"))
217                 )
218             );
219             session.execute();
220         } catch (SQLException ex) {
221             throw new IOException(ex);
222         }
223     }
224     /**
225      * With this table, that has given primary keys.
226      * @param table Table name
227      * @param keys Primary keys
228      * @param attrs Attributes
229      * @return New data, modified
230      * @throws IOException If fails
231      */
232     public MkData with(final String table, final String[] keys,
233         final String[] attrs) throws IOException {
234         if (keys.length == 0) {
235             throw new IllegalArgumentException(
236                 String.format("empty list of keys for %s table", table)
237             );
238         }
239         final StringBuilder sql = new StringBuilder("CREATE TABLE ")
240             .append(table).append(" (");
241         Joiner.on(',').appendTo(
242             sql,
243             Iterables.transform(Arrays.asList(keys), H2Data.CREATE_KEY)
244         );
245         if (attrs.length > 0) {
246             sql.append(',');
247             Joiner.on(',').appendTo(
248                 sql,
249                 Iterables.transform(Arrays.asList(attrs), H2Data.CREATE_ATTR)
250             );
251         }
252         sql.append(')');
253         try {
254             new JdbcSession(this.connection()).sql(sql.toString()).execute();
255         } catch (SQLException ex) {
256             throw new IOException(ex);
257         }
258         return this;
259     }
260     /**
261      * Make data source.
262      * @return Data source for JDBC
263      * @throws SQLException If fails
264      */
265     private Connection connection() throws SQLException {
266         return new Driver().connect(this.jdbc, new Properties());
267     }
