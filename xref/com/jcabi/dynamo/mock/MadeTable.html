44  /**
45   * A Table that can be made and dropped.
46   *
47   * <p>Use it like this in your integration test:
48   *
49   * <pre> public class FooITCase {
50   *   private Region region;
51   *   private MadeTable table;
52   *   &#64;Before
53   *   public void prepare() {
54   *     this.region = new Region.Simple(..your IT credentials..);
55   *     this.table = new MadeTable(this.region, new CreateTableRequest()...);
56   *     this.table.createIfAbsent();
57   *   }
58   *   &#64;After
59   *   public void dropTable() {
60   *     this.table.drop();
61   *   }
62   *   &#64;Test
63   *   public void createsAndDeletesItems() {
64   *     Foo foo = new Foo(this.region);
65   *     foo.doSomething();
66   *   }
67   * }</pre>
68   *
69   * <p>In this example, a new DynamoDB table will be created before every
70   * test method, and dropped when it's finished. This may be not the best
71   * approach performance wise, since every table creation takes at least
72   * ten seconds (at the time of writing). To speed things up a little, you
73   * can create table before the entire test case and drop when all methods
74   * are completed:
75   *
76   * <pre> public class FooITCase {
77   *   private static Region region;
78   *   private static MadeTable table;
79   *   &#64;BeforeClass
80   *   public static void prepare() {
81   *     FooITCase.region = new Region.Simple(..your IT credentials..);
82   *     FooITCase.table = new MadeTable(
83   *       FooITCase.region, new CreateTableRequest()...
84   *     );
85   *     FooITCase.table.createIfAbsent();
86   *   }
87   *   &#64;AfterClass
88   *   public static void dropTable() {
89   *     FooITCase.table.drop();
90   *   }
91   *   &#64;Test
92   *   public void createsAndDeletesItems() {
93   *     Foo foo = new Foo(FooITCase.region);
94   *     foo.doSomething();
95   *   }
96   * }</pre>
97   *
98   * <p>You IAM user policy would look like this (XXXXX should be replaced
99   * by your AWS account number):
100  *
101  * <pre>{
102  *   "Statement": [
103  *     {
104  *       "Action": "dynamodb:*",
105  *       "Effect": "Allow",
106  *       "Resource": "arn:aws:dynamodb:us-east-1:XXXXX:table/test-*"
107  *     },
108  *     {
109  *       "Action": "dynamodb:ListTables",
110  *       "Effect": "Allow",
111  *       "Resource": "*"
112  *     }
113  *   ]
114  * }</pre>
115  *
116  * @author Yegor Bugayenko (yegor@tpc2.com)
117  * @version $Id$
118  * @since 0.8
119  */
120 public final class MadeTable {
121 
122     /**
123      * Region.
124      */
125     private final transient Region region;
126 
127     /**
128      * Table name.
129      */
130     private final transient CreateTableRequest request;
131 
132     /**
133      * Public ctor.
134      * @param reg Region
135      * @param req Request
136      */
137     public MadeTable(final Region reg, final CreateTableRequest req) {
138         this.region = reg;
139         this.request = req;
140     }
141 
142     /**
143      * Create table if it's absent.
144      * @throws InterruptedException If something fails
145      * @since 0.9
146      */
147     public void createIfAbsent() throws InterruptedException {
148         if (!this.exists()) {
149             this.create();
150         }
151     }
152 
153     /**
154      * Create table.
155      * @throws InterruptedException If something fails
156      */
157     public void create() throws InterruptedException {
158         final AmazonDynamoDB aws = this.region.aws();
159         final String name = this.request.getTableName();
160         aws.createTable(this.request);
161         Logger.info(this, "DynamoDB table '%s' creation requested...", name);
162         final DescribeTableRequest req = new DescribeTableRequest()
163             .withTableName(name);
164         while (true) {
165             final DescribeTableResult result = aws.describeTable(req);
166             if ("ACTIVE".equals(result.getTable().getTableStatus())) {
167                 Logger.info(
168                     this,
169                     "DynamoDB table '%s' is %s",
170                     name, result.getTable().getTableStatus()
171                 );
172                 break;
173             }
174             TimeUnit.SECONDS.sleep((long) Tv.TEN);
175             Logger.info(
176                 this,
177                 "waiting for DynamoDB table '%s': %s",
178                 name, result.getTable().getTableStatus()
179             );
180         }
181     }
182 
183     /**
184      * Drop table.
185      * @throws InterruptedException If something fails
186      */
187     public void drop() throws InterruptedException {
188         final AmazonDynamoDB aws = this.region.aws();
189         final String name = this.request.getTableName();
190         aws.deleteTable(new DeleteTableRequest().withTableName(name));
191         Logger.info(this, "DynamoDB table '%s' deletion requested", name);
192         while (this.exists()) {
193             Logger.info(this, "DynamoDB table '%s' still exists", name);
194             TimeUnit.SECONDS.sleep((long) Tv.TEN);
195         }
196         Logger.info(this, "DynamoDB table '%s' deleted", name);
197     }
198 
199     /**
200      * The table exists?
201      * @return TRUE if it exists in DynamoDB
202      * @since 0.9
203      */
204     public boolean exists() {
205         final AmazonDynamoDB aws = this.region.aws();
206         final String name = this.request.getTableName();
207         boolean exists;
208         try {
209             aws.listTables(
210                 new ListTablesRequest()
211                     .withExclusiveStartTableName(name)
212                     .withLimit(1)
213             );
214             exists = true;
215             Logger.info(this, "DynamoDB table '%s' already exists", name);
216         } catch (ResourceNotFoundException ex) {
217             exists = false;
218             Logger.info(this, "DynamoDB table '%s' doesn't exist", name);
219         }
220         return exists;
221     }
